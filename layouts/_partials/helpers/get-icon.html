{{- $name := .name -}} {{- $type := .type -}} {{- $iconsConfig := index site.Params.SineDie.Icons $type | default dict -}} {{- $errors := slice }} {{- if not $iconsConfig }} {{- $errors = $errors | append "Missing config" }} {{- end -}} {{- $errorLevel := $iconsConfig.ErrorLevel | default "warning" -}} {{- $useDefaultIcon := $iconsConfig.UseDefaultIcon -}} {{- $defaultVisualSign := "" -}} {{- if eq $useDefaultIcon "icon" -}} {{- $defaultVisualSign = $iconsConfig.DefaultIcon -}} {{- else if eq $useDefaultIcon "name" -}} {{- $defaultVisualSign = $name -}} {{- end -}} {{- $visualSign := "" -}} {{- if reflect.IsMap $iconsConfig.Entries -}} {{- $visualSign = index $iconsConfig.Entries $name -}} {{- else if in ($iconsConfig.Entries | default slice) $name -}} {{- $visualSign = $name -}} {{- end -}} {{- if and (not $visualSign) (in (slice "icon" "name") $useDefaultIcon) }} {{- $errors = $errors | append "Missing icon" }} {{- end }} {{- if $errors | len }} {{- $caller := templates.Current.Parent.Name }} {{- if eq page.Kind "page" }} {{ with page.File }}{{ $caller = .Path }}{{ end }} {{- end }} {{- if $iconsConfig }} {{- $errors = $errors | append (printf "Config: %v" $iconsConfig) }} {{- end -}} {{- $errors = $errors | append (printf "Caller: %v" $caller) }} {{- $error := delimit $errors "\n " }} {{- $error = printf "Icon name '%v' type '%v'\n %s" $name $type $error -}} {{- if eq $errorLevel "error" }} {{- errorf $error }} {{- else if or (eq $errorLevel "warning") site.Params.SineDie.Debug }} {{- warnf $error }} {{- end }} {{- end }} {{- return $visualSign -}}
