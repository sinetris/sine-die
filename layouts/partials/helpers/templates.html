{{- /* Renders the given html entity attributes. Accepts only named parameters. @param {dict} attributes A Map of the attributes to render @param {slice} [exclude] A list of attributes keys to exclude from the rendering @example This code: ```go-html-template {{ $attributes := dict }} {{ $attributes := merge $attributes (dict "class" (slice "try-me" "centered")) }} {{ $attributes := merge $attributes (dict "inert" true "hidden" false) }} {{ $attributes := merge $attributes (dict "draggable" "true") }} {{ $attributes := dict "src" "images/logo.png" "alt" "The Logo" "caption"Using [CommonMark](https://commonmark.org/)" }} {{ $exclude := (slice "src" "alt" "caption") }} ``` Will render: ```html ``` */ -}} {{ define "render-attributes" -}} {{- with $attributes := .attributes -}} {{- if reflect.IsMap $attributes -}} {{- $excludeAttributes := .exclude | default dict }} {{- range $key, $value := $attributes }} {{- if and (not (in $excludeAttributes $key)) (not (eq $value false)) }} {{- if eq $value true -}} {{- printf " %s" $key | safeHTMLAttr -}} {{- else -}} {{- if reflect.IsSlice $value -}} {{- $value = delimit $value " " -}} {{- end -}} {{- printf " %s=%q" $key $value | safeHTMLAttr -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
